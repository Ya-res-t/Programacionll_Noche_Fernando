#include <iostream>
#define _USE_MATH_DEFINES
#include <math.h>

{

        {
        case 1:
        {
            for (int i = 0; i < 7; i++) {
                for (int j = 0; j < 7; j++)
                {
                    if (i == 0 or j == 0 or i == 6)
                        cout << "* ";
                    else
                        cout << "  ";
                }
                cout << endl;
            }
            break;

        }
        case 2:
        {
            for (int i = 0; i < 7; i++) {
                for (int j = 0; j < 7; j++)
                {
                    if (i == 3 or j == 0 or j == 6)
                        cout << "* ";
                    else
                        cout << "  ";
                }
                cout << endl;
            }

        }
        case 3:
        {
            for (int i = 0; i < 7; i++) {
                for (int j = 0; j < 7; j++)
                {
                    if (i == 0 or j == 0 or i == 6 or i == 3 && j >= 3 or j == 6 && i >= 3)
                        cout << "* ";
                    else
                        cout << "  ";
                }
                cout << endl;
            }

        }
        case 4:
        {
            for (int i = 0; i < 7; i++) {
                for (int j = 0; j < 7; j++)

                    if (i == 0 or j == 0 or i == 6)

                    if (i == 0 or j == 0 or i == 6 or i == 3 && j >= 3 or j == 6 && i >= 3)

                    else
                        cout << "  ";
                }
                cout << endl;
            }
            cout << endl;

}
